Please place the downloaded files to the folder ./R_metagene_TPR1/bam_bai/TPR1/

"TPR1_Col_ChIP.bam"
"TPR1_Col_ChIP.bam.bai"
"TPR1_Col_input.bam"
"TPR1_Col_input.bam.bai"
"TPR1_eds1_ChIP.bam"
"TPR1_eds1_ChIP.bam.bai"
"TPR1_eds1_input.bam"
"TPR1_eds1_input.bam.bai"
